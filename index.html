void insert()
{
if(n < max){
printf("\nEnter the position where you wish to insert new element\n");
scanf("%d",&pos);
if(pos>=n+1)
printf("Invalid position\n");
else
{
printf("\n Enter the element to be inserted:\t");
scanf("%d",&ele);
for(i=n-1;i>=pos;i--)
a[i+1]=a[i];
a[pos]=ele;
n++;
}
} else {
printf("array is full");
}
}
void delete()
{
printf("\nEnter the position from where you wish to delete element\n");
scanf("%d",&pos);
if(pos>=n+1)
printf("Invalid position\n");
else
{
printf("\n Deleted Element is : %d",a[pos]);
for(i=pos;i< n-1;i++)
a[i]=a[i+1];
n--;
}
}

{

int priority(char symb)
{
int p;
switch(symb)
{
case '+':
case '-': p=1;
break;
case '*':
case '/':
case '%': p=2;
break;
case '^':
case '$': p=3;
break;
case '(':
case ')': p=0;
break;
case '#':p=-1;
break;
}
return p;
}

double Compute(char symbol, int op1, int op2)
{
switch(symbol)
{
case '+' : return op1 + op2;
case '-' : return op1 - op2;
case '*' : return op1 * op2;
case '/' : return op1 / op2;
case '%' : return (op1% op2);
case '^' : return (int)pow(op1,op2);
default : return 0;
}
}

{
int s[20], op1, op2;

s[++top] = symbol -'0';
